        // ==========================================
        // 一物一码核心逻辑：从网址参数读取数据
        // ==========================================
        function getUrlParam(paramName) {
            const urlParams = new URLSearchParams(window.location.search);
            return urlParams.get(paramName);
        }

        // 尝试从网址参数里获取溯源数据，如果没有，则使用默认值
        // 示例网址：https://你的域名/?orchard=黄良华&date=2026-10-12&se=0.12&sugar=18:1&id=XMTX-A001
        const orchard = getUrlParam('orchard') || '黄良华果园 · 3号地块';
        const farmer  = getUrlParam('farmer')  || '黄良华';
        const date    = getUrlParam('date')    || '2026年10月12日';
        const se      = getUrlParam('se')      || '0.12 mg/kg';
        const sugar   = getUrlParam('sugar')   || '18:1';
        const traceId = getUrlParam('id')      || 'XMTX-A001';

        // 构建溯源信息数组（标签名 + 数值）
        const infoData = [
            { label: '🍑 产品名称', value: '贵长猕猴桃（特级精选）' },
            { label: '📍 产地果园', value: orchard },
            { label: '👨‍🌾 种植果农', value: farmer },
            { label: '📅 采摘日期', value: date },
            { label: '🧪 硒含量',    value: se + ' <span class="badge">富硒认证</span>' },
            { label: '🍬 黄金甜酸比', value: sugar },
            { label: '🌱 种植方式',   value: '绿肥翻压 · 林下养蚯蚓' },
            { label: '🆔 溯源编码',   value: traceId }
        ];

        // 将上述数据渲染到页面中
        const container = document.getElementById('info-container');
        infoData.forEach(item => {
            const div = document.createElement('div');
            div.className = 'info-item';
            div.innerHTML = `
                <span class="info-label">${item.label}</span>
                <span class="info-value highlight">${item.value}</span>
            `;
            container.appendChild(div);
        });
